```

    "country_code":"DE",
    "zipcode":"55252",
    "place":"Mainz-Kastel",
    "state":"Hessen",
    "state_code":"HE",
    "province":"Regierungsbezirk Darmstadt",
    "province_code":"064",
    "community":"Wiesbaden,
     Landeshauptstadt",
     "community_code":"06414",
    "latitude":"50.0102",
    "longitude":"8.2844"
  }
```
